//Scrivere un programma che legge tre valori da standard input e ne stampi la media aritmetica.

#include <stdio.h>

float agv(int x, int y, int z){
    float agv = 0;
    agv = x + y + z ;
    agv = agv / 3;
    return agv;
}

int main(){
    int x, y, z;
    printf("inserici il primo valore : ");
    scanf("%02d", x);
    printf("\n inserisci il secondo valre : ");
    scanf("%02d", y);
    printf("\n inserisci il terzo valore : ");
    scanf("%02d",z);
    float media;
    media = agv (x,y,z);
    printf("La media dei valori inseriti vale: %f", media);
}
